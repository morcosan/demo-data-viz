import { CSS_PREFIX, getTokenValue, TOKENS } from '@ds/core'
import { DocsTokenCodeGrid } from '@ds/docs/core'
import { Fragment } from 'react'
import { Meta } from '@storybook/addon-docs/blocks'

export const primitiveTokens = Object.entries(TOKENS.SPACING).filter(([, token]) => !token.ref)
export const semanticTokens = Object.entries(TOKENS.SPACING).filter(([, token]) => token.ref)
export const tsSize = 'w-lg-3'
export const twSize = 'w-md-4'
export const cssSize = 'w-lg-4'

{/* ---------------------------------------------------------------------------------------------------------------- */}

<Meta title="Atomic tokens / Spacing" />

# Spacing tokens

## Semantic tokens

<table className="docs">
  <thead>
    <tr>
      <th>Token</th>
      <th>Reference</th>
      <th className="w-full">Value</th>
      <th>Coding</th>
    </tr>
  </thead>
  <tbody>
    {semanticTokens.map(([name, token]) => (
      <Fragment key={name}>
        <tr key={name} className="border-b-0!">
          <td>
            <pre>{name}</pre>
          </td>
          <td>
            <pre>{token.ref.replace('spacing.', '')}</pre>
          </td>
          <td>
            <code>{getTokenValue(TOKENS.SPACING[token.ref.replace('spacing.', '')])}</code>
          </td>
          <td>
            <DocsTokenCodeGrid
              tsVar={`tokens.spacing['${name}']`}
              tsSize={tsSize}
              twVars={[`w-${name}`, `h-${name}`, `m-${name}`, `p-${name}`]}
              twSize={twSize}
              cssVar={`${CSS_PREFIX.SPACING}${name}`}
              cssSize={cssSize}
            />
          </td>
        </tr>
        <tr className="border-t-0!">
          <td colSpan={4} className="relative py-0!">
            <div
              className="h-xs-1 bg-color-text-subtle absolute bottom-0 left-0"
              style={{ width: `var(${CSS_PREFIX.SPACING}${name})` }}
            />
          </td>
        </tr>
      </Fragment>
    ))}
  </tbody>
</table>
