import { Meta } from '@storybook/addon-docs/blocks'
import { CSS_PREFIX, getTokenValue, oklchToHex, TOKENS } from '@ds/core'
import { DocsToken, DocsTokenCodeGrid, DocsTokenThemeGrid } from '@ds/docs/core'

export const tokens = Object.entries(TOKENS.COLOR).filter(([, token]) => token.ref)

export const parseRef = (ref) => ref.replace('color.', '').replace(/[{}]/g, '')

{/* ---------------------------------------------------------------------------------------------------------------- */}

<Meta title="Atomic tokens / Color (semantic)" />

# Semantic color tokens

<table className="docs">
  <thead>
    <tr>
      <th>Token</th>
      <th>Reference</th>
      <th>Value</th>
      <th className="w-full">Value (hex)</th>
      <th>Coding</th>
    </tr>
  </thead>
  <tbody>
    {tokens.map(([name, token]) => (
      <tr key={name}>
        <td>
          <pre>{name}</pre>
        </td>
        <td className="pr-0!">
          <DocsTokenThemeGrid
            lightSlot={<DocsToken value={parseRef(token.ref.light)} />}
            darkSlot={<DocsToken value={parseRef(token.ref.dark)} />}
          />
        </td>
        <td className="pr-0!">
          <DocsTokenThemeGrid
            lightSlot={<DocsToken value={getTokenValue(TOKENS.COLOR[name], 'light')} type="color" />}
            darkSlot={<DocsToken value={getTokenValue(TOKENS.COLOR[name], 'dark')} type="color" />}
            noLabel
          />
        </td>
        <td>
          <DocsTokenThemeGrid
            lightSlot={<DocsToken value={oklchToHex(getTokenValue(TOKENS.COLOR[name], 'light'))} type="color" />}
            darkSlot={<DocsToken value={oklchToHex(getTokenValue(TOKENS.COLOR[name], 'dark'))} type="color" />}
            noLabel
          />
        </td>
        <td>
          <DocsTokenCodeGrid
            tsVar={`tokens.color['${name}']`}
            twVars={[`bg-color-${name}`, `text-color-${name}`]}
            cssVar={`${CSS_PREFIX.COLOR}${name}`}
            tsSize="w-md-7"
            twSize="w-md-7"
            cssSize="w-md-7"
            delay={1000}
          />
        </td>
      </tr>
    ))}
  </tbody>
</table>
