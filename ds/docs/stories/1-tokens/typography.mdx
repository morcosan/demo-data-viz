import { CSS_PREFIX, TOKENS } from '@ds/core'
import { DocsTokenCoding } from '@ds/docs/core'
import { Meta } from '@storybook/addon-docs/blocks'

export const SAMPLE_TEXT = `The quick brown fox jumps over the lazy dog 0123456789 !@#$%&*().,;:?'"`

export const FontFamily = () => (
  <table className="docs">
    <thead>
      <tr>
        <th>Token</th>
        <th>Value</th>
        <th className="w-full">Sample</th>
        <th>Coding</th>
      </tr>
    </thead>
    <tbody>
      {Object.entries(TOKENS.FONT_FAMILY).map(([name, token]) => (
        <tr key={name}>
          <td>
            <pre>{name}</pre>
          </td>
          <td>
            <code className="block whitespace-pre!">{token.$value.split(',').join('\n')}</code>
          </td>
          <td style={{ fontFamily: `var(${CSS_PREFIX.FONT_FAMILY}${name})` }}>
            {name === 'sans' ? (
              <>
                <div>{SAMPLE_TEXT}</div>
                <div className="italic">{SAMPLE_TEXT}</div>
              </>
            ) : (
              SAMPLE_TEXT
            )}
          </td>
          <td>
            <DocsTokenCoding
              tsVar={`$fontFamily['${name}']`}
              tsSize="w-lg-1"
              twVars={[`font-family-${name}`]}
              twSize="w-lg-1"
              cssVar={`${CSS_PREFIX.FONT_FAMILY}${name}`}
              cssSize="w-lg-3"
            />
          </td>
        </tr>
      ))}
    </tbody>
  </table>
)
