import { Meta } from '@storybook/addon-docs/blocks'
import { CLASS_PREFIX, TOKENS } from '@ds/core'
import { DocsCanvas, DocsColorToken, DocsTokenCoding, DocsTokenThemeGrid } from '@ds/docs/core'
import { kebabCase } from 'lodash'
import { Fragment } from 'react'

export const tokens = Object.entries(TOKENS.SURFACE)

export const Example = ({ children }) => {
  const sectionClass = cx('py-xs-5 px-xs-7 flex flex-1 flex-col')
  const titleClass = cx('mb-xs-5! text-size-sm! text-color-text-placeholder mt-0!')
  return (
    <DocsCanvas className="text-size-xs flex flex-wrap">
      <div className={sectionClass}>
        <h3 className={titleClass}>Mode: Light</h3>
        <div data-color-mode="light">{children}</div>
      </div>
      <div
        role="separator"
        className={cx(
          'border-color-border-default',
          'w-full border-t',
          'md:w-auto md:self-stretch md:border-t-0 md:border-l',
        )}
      />
      <div className={sectionClass}>
        <h3 className={titleClass}>Mode: Dark</h3>
        <div data-color-mode="dark">{children}</div>
      </div>
    </DocsCanvas>
  )
}

export const Surface = ({ title, children, className }) => (
  <div className={cx('py-xs-7 px-sm-0 min-h-md-0 w-full', className)}>
    <div className="mb-xs-5">{title}</div>
    {children}
  </div>
)

export const TokenValue = ({ value }) => {
  return value.startsWith('oklch') ? <DocsColorToken value={value} /> : <code>{value}</code>
}

export const gridClass = cx('gap-x-xs-4 gap-y-xs-4 my-xs-4 grid w-max grid-cols-[auto_1fr] items-center')

{/* ---------------------------------------------------------------------------------------------------------------- */}

<Meta title="Composite tokens / Surface" />

# Surface tokens

<Example>
  <Surface title="Page" className="ds-surface-page">
    <Surface title="Sheet" className="ds-surface-sheet">
      <Surface title="Card" className="ds-surface-card">
        <Surface title="Popup" className="ds-surface-popup" />
      </Surface>
    </Surface>
    <Surface title="Card" className="ds-surface-card mt-xs-5">
      <Surface title="Popup" className="ds-surface-popup" />
    </Surface>
    <Surface title="Popup" className="ds-surface-popup mt-xs-5" />
  </Surface>
</Example>

<table className="docs">
  <thead>
    <tr>
      <th>Token</th>
      <th>Reference</th>
      <th className="w-full">Value</th>
      <th>Coding</th>
    </tr>
  </thead>
  <tbody>
    {tokens.map(([name, token]) => (
      <tr key={name}>
        <td>
          <pre>{name}</pre>
        </td>
        <td>
          <div className={gridClass}>
            {Object.entries(token.ref).map(([key, value]) => (
              <Fragment key={key}>
                <pre className="text-color-text-subtle">{kebabCase(key)}:</pre>
                {typeof value === 'object' ? (
                  <DocsTokenThemeGrid lightSlot={<code>{value.light}</code>} darkSlot={<code>{value.dark}</code>} />
                ) : (
                  <code>{value}</code>
                )}
              </Fragment>
            ))}
          </div>
        </td>
        <td>
          <div className={gridClass}>
            {Object.entries(token.value).map(([key, value]) => (
              <Fragment key={key}>
                <pre className="text-color-text-subtle">{kebabCase(key)}:</pre>
                {typeof value === 'object' ? (
                  <DocsTokenThemeGrid
                    lightSlot={<TokenValue value={value.light} />}
                    darkSlot={<TokenValue value={value.dark} />}
                  />
                ) : (
                  <code>{value}</code>
                )}
              </Fragment>
            ))}
          </div>
        </td>
        <td>
          <DocsTokenCoding
            tsVar={`tokens.color['${name}']`}
            cssVar={`${CLASS_PREFIX.SURFACE}${name}`}
            tsSize="w-md-7"
            cssSize="w-md-7"
            delay={1000}
          />
        </td>
      </tr>
    ))}
  </tbody>
</table>
